Testes de unidade

Pequenos trechos de código (por programadores para programadores!), escritos na mesma linguagem do produto, com o objetivo de testar a menor unidade do software (um método ou função) 

class Calculator {
    public function sum($first, $second) {
        return $first + $second;
    }

    public function subtract($first, $second) {
        /* ... */
}

class CalculatorTest {
    public function sumShouldReturnTheSumOfTwoNumbers() {
        $calculator = new Calculator();
        $result = $calculator->sum(1, 2);
        if ($result !== 3) {
            throw new Exception("Something wrong is not right... ¯\_(ツ)_/¯");
        }
    }

    public function subtractShouldReturnTheDifferenceOfTwoNumbers() {
        /* ... */
}
